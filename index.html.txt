async function callAPI(userMsg) {
  const res = await fetch(PROXY_URL, {
    method:'POST',
    headers:{'Content-Type':'application/json'},
    body:JSON.stringify({messages:[{role:'user',content:userMsg}]})
  });
  if(!res.ok) throw new Error('API error '+res.status+' — check your Worker URL and API key');
  const data = await res.json();
  const text = (data.content||[]).filter(b=>b.type==='text').map(b=>b.text).join('');
  const m = text.match(/```json\s*([\s\S]*?)```/);
  if(!m) throw new Error('No JSON returned — unexpected model response');
  return JSON.parse(m[1].trim());
}

    updHistory(); updCompare();
  }catch(e){
    errEl.textContent='Error: '+e.message;
    errEl.style.display='block';
  }
  document.getElementById('l_waiting').classList.remove('show');
  btn.disabled=false; btn.textContent='Run Ceribus forecast →';
}

function renderResult(r,px,subject,metric,unit){
  document.getElementById(px+'rs').textContent=subject;
  document.getElementById(px+'rml').textContent=metric+(unit?' ('+unit+')':'');
  document.getElementById(px+'rv').textContent=r.point_estimate||'—';
  const low=pn(r.range_low), high=pn(r.range_high), point=pn(r.point_estimate);
  const sn=px.replace('_','');
  if(!isNaN(low)&&!isNaN(high)&&!isNaN(point)){
    RS[sn]={low,high,point,pl:r.point_estimate};
    const span=high-low||1, extLow=low-span*0.3, extHigh=high+span*0.3, extSpan=extHigh-extLow;
    const pPct=((point-extLow)/extSpan)*100, fL=((low-extLow)/extSpan)*100, fR=((high-extLow)/extSpan)*100;
    document.getElementById(px+'rfill').style.left=fL+'%'; document.getElementById(px+'rfill').style.width=(fR-fL)+'%';
    document.getElementById(px+'rpoint').style.left=pPct+'%';
    document.getElementById(px+'apoint').style.display='none';
    document.getElementById(px+'rl').textContent=r.range_low; document.getElementById(px+'rh').textContent=r.range_high;
    document.getElementById(px+'legend').innerHTML='<div class="legend-item"><div class="legend-dot" style="background:var(--acc)"></div>Forecast: '+r.point_estimate+'</div>';
    document.getElementById(px+'rangeRow').style.display='block';
    document.getElementById(px+'actualStrip').style.display='flex';
    document.getElementById(px+'actualInput').value='';
    document.getElementById(px+'accCallout').className='acc-callout';
  }
  const conf=Math.round(r.confidence||0);
  document.getElementById(px+'cf').style.width=conf+'%'; document.getElementById(px+'cp').textContent=conf+'%';
  document.getElementById(px+'cn').textContent=confNote(conf);
  const b=document.getElementById(px+'rb');
  if(conf>=70){b.textContent='High confidence';b.className='rbadge bhi';}
  else if(conf>=45){b.textContent='Moderate confidence';b.className='rbadge bmid';}
  else{b.textContent='Low confidence';b.className='rbadge blo';}
  if(r.factors&&r.factors.length)
    document.getElementById(px+'fg').innerHTML=r.factors.map(f=>{
      const c=f.direction==='positive'?'var(--acc)':f.direction==='negative'?'var(--danger)':'var(--warn)';
      return`<div class="factor"><span class="fn">${f.name}</span><div class="fbw"><div class="fb" style="width:${Math.round(f.impact)}%;background:${c}"></div></div><span class="fv">${Math.round(f.impact)}</span></div>`;
    }).join('');
  document.getElementById(px+'an').textContent=r.analysis||'';
  document.getElementById(px+'rk').textContent=r.key_risks?'Risk: '+r.key_risks:'';
  document.getElementById(px+'result').classList.add('show');
}

function applyActual(sn){
  const px=sn+'_', input=document.getElementById(px+'actualInput').value.trim(); if(!input)return;
  const actual=pn(input), store=RS[sn]; if(isNaN(actual))return;
  const {low,high,point,pl}=store, span=high-low||1;
  const extLow=low-span*0.3, extHigh=high+span*0.3, extSpan=extHigh-extLow;
  const ca=Math.min(Math.max(actual,extLow),extHigh);
  const pPct=((point-extLow)/extSpan)*100, aPct=((ca-extLow)/extSpan)*100;
  const fL=((low-extLow)/extSpan)*100, fR=((high-extLow)/extSpan)*100;
  document.getElementById(sn+'_rfill').style.left=fL+'%'; document.getElementById(sn+'_rfill').style.width=(fR-fL)+'%';
  document.getElementById(sn+'_rpoint').style.left=pPct+'%';
  const ap=document.getElementById(sn+'_apoint'); ap.style.display='block'; ap.style.left=aPct+'%';
  document.getElementById(sn+'_legend').innerHTML=`<div class="legend-item"><div class="legend-dot" style="background:var(--acc)"></div>Forecast: ${pl}</div><div class="legend-item"><div class="legend-dot" style="background:var(--acc2)"></div>Actual: ${input}</div>`;
  const diff=Math.abs(actual-point), pct=point?Math.round((diff/Math.abs(point))*100):0;
  const callout=document.getElementById(sn+'_accCallout');
  let cls,icon,msg;
  if(actual>=low&&actual<=high){if(pct<=10){cls='a-great';icon='◎';msg=`Spot on — actual within ${pct}% of forecast and inside the range.`;}else{cls='a-good';icon='◉';msg=`Good call — actual inside the forecast range (${pct}% off the point estimate).`;}}
  else if(pct<=25){cls='a-ok';icon='○';msg=`Close but outside the range — actual was ${pct}% off the point estimate.`;}
  else{cls='a-off';icon='◌';msg=`Missed — actual was ${pct}% off and outside the range. Consider adjusting your variable set.`;}
  callout.className='acc-callout show '+cls;
  callout.innerHTML=`<span class="acc-icon">${icon}</span><span class="acc-text">${msg}</span>`;
}
